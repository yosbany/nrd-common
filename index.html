Ejemplo de Uso

<!-- Incluir desde CDN -->
<script src="https://cdn.jsdelivr.net/gh/yosbany/nrd-common@main/dist/nrd-common.js"></script>

<script>
  // Crear logger
  window.logger = new NRDCommon.Logger('MiApp');
  
  // Inicializar NRD
  await NRDCommon.initializeNRD();
  
  // Crear AuthService
  const authService = new NRDCommon.AuthService(window.nrd);
  
  // Usar utilidades
  const formatted = NRDCommon.formatCurrency(1234.56);
  NRDCommon.showSuccess('Éxito', 'Operación completada');
</script>
